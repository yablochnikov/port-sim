private createBoat() {
        const boat = new Graphics();
        boat.beginFill(0x663300); // Adjust boat color as needed
        boat.drawRect(0, 0, this.boatWidth, this.boatHeight);
        boat.endFill();

        const randomPierIndex = Math.floor(Math.random() * this.piers.length);
        const targetPier = this.piers[randomPierIndex];

        const boatX = this.app.screen.width; // Start boat outside the port
        const boatY = targetPier.y + targetPier.height / 2 - this.boatHeight / 2;
        boat.x = boatX;
        boat.y = boatY;

        const boatSpeed = 5; // Adjust boat speed as needed

        this.app.stage.addChild(boat);

        const moveBoat = () => {
            boat.x -= boatSpeed;

            if (boat.x <= targetPier.x + targetPier.width) {
                clearInterval(moveBoatInterval);

                // Simulate loading (visual or other logic)
                boat.beginFill(0xff0000); // Change color to represent loading
                boat.endFill();
                setTimeout(() => {
                    boat.beginFill(0x663300); // Restore original color
                    boat.endFill();

                    // Move boat back out of the port
                    const moveBoatBack = () => {
                        boat.x += boatSpeed;
                        if (boat.x >= this.app.screen.width) {
                            clearInterval(moveBoatBackInterval);
                            this.app.stage.removeChild(boat);
                        }
                    };
                    const moveBoatBackInterval = setInterval(moveBoatBack, 10); // Adjust return speed as needed
                }, 2000);
            }
        };

        const moveBoatInterval = setInterval(moveBoat, 10);
    }
